### 普通结构体
```
type info struct {
	NAME string
	QQ_ID int
	Email string
}

// 结构体
func main()  {
	user_info:=new(info)

	user_info.NAME = "姚波海"
	user_info.QQ_ID = 2695476342
	user_info.Email = "2695476342@qq.com"

	fmt.Println("用户姓名:",user_info.NAME,"联系QQ:",user_info.QQ_ID,"电子邮箱:",user_info.Email,"当前程序内存地址:",user_info)
}
```
